\begin{itemize}[leftmargin=*]

    \item \textbf{Free PPNs list (allocated in order)}:  $\mathtt{0x3}$,  $\mathtt{0x9}$,  $\mathtt{0x12}$,  $\mathtt{0x19}$

\end{itemize}
